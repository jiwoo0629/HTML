o <img> 태그 : 이미지
 
- 속성
 src = "경로명(gif, jpeg)" : 이미지 파일 경로
 align = "문자열" : 이미지 정렬 방식
  top / bottom / left / right / middle
 alt = "문자열" : 이미지 설명 지정
 width = "숫자" : 이미지 폭 지정
 height = "숫자" : 이미지 높이 지정
 border = "숫자" : 이미지 테두리 두께 지정
 hspace(vspace) = "숫자" : 이미지 좌우(상하) 여백 지정

o <a> 태그 : 하이퍼링크
 <a> ~ </a> <br> 꼴로 사용
 
- 하이퍼링크란? 연결된 개체
  여러 html 문서를 링크로 연결해 이동 가능
 
- 속성
 href = "주소" : 이동하고자 하는 주소 지정
 name = "문자열" : 현재 지정된 위치로 링크 설정
 target = "_blank" : 링크 새로운 창으로 열기
 title = "문자열" : 하이퍼링크 설명
 
- 색상 지정 (<body> 태그에 명시)
 link = "색상" : 사용자가 선택하지 않은 상태
 vlink = "색상" : 사용자가 한번이라도 방문했던 상태
 alink = "색상" : 사용자가 마우스로 누르고 있는 상태
